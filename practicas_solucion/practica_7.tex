\section*{El motor de búsqueda de prolog}

\subsection{Ejercicio 1}
\paragraph{Base de conocimiento:}
\begin{centrado}
\begin{minted}{prolog}
padre(juan, carlos).
padre(juan, luis).
padre(carlos, daniel).
padre(carlos, diego).

padre(luis, pablo).
padre(luis, manuel).
padre(luis, ramiro).
abuelo(X,Y) :- padre(X,Z), padre(Z,Y).
\end{minted}
\end{centrado}

\paragraph{I.} La consulta \mintinline{prolog}{abuelo(X, manuel)}, devuele \mintinline{prolog}{X = juan}.
